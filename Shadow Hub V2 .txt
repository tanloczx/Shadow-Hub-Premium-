local function CreateBackpackESP(Player)
	if Player == LP then return end
	
	local Billboard = Instance.new("BillboardGui")
	Billboard.Size = UDim2.new(0,200,0,40)
	Billboard.AlwaysOnTop = true
	Billboard.Parent = ESPFolder
	
	local Text = Instance.new("TextLabel", Billboard)
	Text.Size = UDim2.new(1,0,1,0)
	Text.BackgroundTransparency = 1
	Text.TextColor3 = Color3.new(0,1,1)
	Text.TextStrokeTransparency = 0
	Text.TextScaled = true
	
	task.spawn(function()
		while BackpackESPEnabled and Player.Parent do
			local Char = Player.Character
			if Char and Char:FindFirstChild("Head") then
				Billboard.Adornee = Char.Head
				local tools = {}
				
				local toolInHand = Char:FindFirstChildOfClass("Tool")
				if toolInHand then 
					table.insert(tools, "[" .. toolInHand.Name .. "]") 
				end
				
				for _,t in pairs(Player.Backpack:GetChildren()) do
					table.insert(tools, t.Name)
				end
				
				if #tools > 0 then
					Text.Text = table.concat(tools,", ")
				else
					Text.Text = "Trống"
				end
				Text.TextSize = ESPSize
			end
			task.wait(1)
		end
		Billboard:Destroy()
	end)
end

            else
                RangeVisual.Color = getgenv().RangeColor
            end
            RangeVisual.Transparency = getgenv().VisualTransparency
        else
            RangeVisual.Transparency = 1
        end
    end
end)

local function SecureMove(targetCF)
    local lp = game.Players.LocalPlayer
    if not lp.Character or not lp.Character:FindFirstChild("HumanoidRootPart") then return end
    local hrp = lp.Character.HumanoidRootPart
    
    if TeleMode == "Teleport" then
        hrp.Velocity = Vector3.new(0, 0, 0)
        hrp.CFrame = targetCF + Vector3.new(0, 5, 0)
    elseif TeleMode == "Tween" then
        local noclipConn
        noclipConn = game:GetService("RunService").Stepped:Connect(function()
            if lp.Character then
                for _, v in pairs(lp.Character:GetDescendants()) do
                    if v:IsA("BasePart") then v.CanCollide = false end
                end
            end
        end)
        
        local distance = (hrp.Position - targetCF.Position).Magnitude
        local speed = 150 
        local tweenInfo = TweenInfo.new(distance / speed, Enum.EasingStyle.Linear)
        local tween = game:GetService("TweenService"):Create(hrp, tweenInfo, {CFrame = targetCF + Vector3.new(0, 5, 0)})
        
        tween.Completed:Connect(function()
            if noclipConn then noclipConn:Disconnect() end
            for _, v in pairs(lp.Character:GetDescendants()) do
                if v:IsA("BasePart") then v.CanCollide = true end
            end
        end)
        tween:Play()
    end
end

                                        end
                                    end
                                end)

                end

local AutoRandom = false
local RollAmount = 1
local KeepCommon = false
local KeepRare = false
local KeepLegendary = true
local KeepMythical = true
local KeepSecret = true

TabFruit:AddButton({
    Name = "Inf Beli",
    Callback = function()
        local args = {{false, {Legendary=true, Common=true, Rare=true, Mythical=true, Secret=false}, -100000000}}
        game:GetService("Players").LocalPlayer:WaitForChild("PlayerGui"):WaitForChild("Random"):WaitForChild("Bright"):FireServer(unpack(args))
    end    
})

TabFruit:AddSection({Name = "Chọn Để Giữ Lại Fruit Lưu Ý chỉ nên giữ Secret nếu Không Không nhắc sớm"})

TabFruit:AddToggle({Name = "Giữ Common", Default = false, Callback = function(v) KeepCommon = v end})
TabFruit:AddToggle({Name = "Giữ Rare", Default = false, Callback = function(v) KeepRare = v end})
TabFruit:AddToggle({Name = "Giữ Legendary", Default = true, Callback = function(v) KeepLegendary = v end})
TabFruit:AddToggle({Name = "Giữ Mythical", Default = true, Callback = function(v) KeepMythical = v end})
TabFruit:AddToggle({Name = "Giữ Secret", Default = true, Callback = function(v) KeepSecret = v end})

TabFruit:AddSection({Name = "Số Lượng Random"})

TabFruit:AddSlider({
    Name = "number fruits",
    Min = 1, Max = 100, Default = 1,
    Increment = 1, ValueName = "Trái",
    Callback = function(v) RollAmount = v end    
})

TabFruit:AddToggle({
    Name = "Auto Random Fruit",
    Default = false,
    Callback = function(v) AutoRandom = v end
})

task.spawn(function()
    while true do
        if AutoRandom then
            pcall(function()
                local args = {
                    {
                        false,
                        {
                            Common = not KeepCommon,
                            Rare = not KeepRare,
                            Legendary = not KeepLegendary,
                            Mythical = not KeepMythical,
                            Secret = not KeepSecret
                        },
                        80000
                    }
                }
                
                for i = 1, RollAmount do
                    if not AutoRandom then break end
                    game:GetService("Players").LocalPlayer:WaitForChild("PlayerGui"):WaitForChild("Random"):WaitForChild("Bright"):FireServer(unpack(args))
                    task.wait(0.1)
                end
            end)
        end
        task.wait(1)
    end
end)
