local function GetHWID() return game:GetService("RbxAnalyticsService"):GetClientId() end

local function SafeHttpGet(url)
    local status, response = pcall(function()
        return HttpService:RequestAsync({
            Url = url,
            Method = "GET",
            Headers = {["Cache-Control"] = "no-cache"}
        })
    end)
    if status and response.Success then
        return response.Body
    end
    return "null"
end

local function CheckKey(inputKey)
    local url = BASE_URL .. "Keys/" .. inputKey .. ".json?t=" .. os.time()
    local result = SafeHttpGet(url)
    if result ~= "null" and result ~= "" then
        local data = HttpService:JSONDecode(result)
        local currentHWID = GetHWID()
        
        if (os.time() * 1000) > (data.expires or 0) then return false, "Key hết hạn rồi đại ca!" end
        
        if data.hwids and data.hwids[currentHWID] then 
            return true, data
        elseif (data.used or 0) < (data.slots or 1) then
            pcall(function()
                local nextUsedValue = (data.used or 0) + 1
                local patchUrl = BASE_URL .. "Keys/" .. inputKey .. ".json?x-http-method-override=PATCH"
                
                local updateTable = {
                    used = nextUsedValue,
                    player_name = Player.Name,
                    hwids = data.hwids or {}
                }
                updateTable.hwids[currentHWID] = true
                
                HttpService:RequestAsync({
                    Url = patchUrl,
                    Method = "POST",
                    Headers = {["Content-Type"] = "application/json"},
                    Body = HttpService:JSONEncode(updateTable)
                })
            end)
            return true, data
        else 
            return false, "Key này đã có người sử dụng rồi, resethwid hoặc tạo key mới đi" 
        end
    end
    return false, "Key không tồn tại trên hệ thống!"
end

	Billboard.AlwaysOnTop = true

	Text.BackgroundTransparency = 1
	Text.TextColor3 = Color3.new(0,1,1)
	Text.TextStrokeTransparency = 0
	Text.TextScaled = true
	
	task.spawn(function()
		while BackpackESPEnabled and Player.Parent do
			local Char = Player.Character
			if Char and Char:FindFirstChild("Head") then
				Billboard.Adornee = Char.Head
				local tools = {}
				
				local toolInHand = Char:FindFirstChildOfClass("Tool")
				if toolInHand then 
					table.insert(tools, "[" .. toolInHand.Name .. "]") 
				end
				
				for _,t in pairs(Player.Backpack:GetChildren()) do
					table.insert(tools, t.Name)
				end
				
				if #tools > 0 then
					Text.Text = table.concat(tools,", ")
				else
					Text.Text = "Trống"
				end
				Text.TextSize = ESPSize
			end
			task.wait(1)
		end
		Billboard:Destroy()
	end)
end

local function SecureMove(targetCF)
    local lp = game.Players.LocalPlayer
    if not lp.Character or not lp.Character:FindFirstChild("HumanoidRootPart") then return end
    local hrp = lp.Character.HumanoidRootPart
    
    if TeleMode == "Teleport" then
        hrp.Velocity = Vector3.new(0, 0, 0)
        hrp.CFrame = targetCF + Vector3.new(0, 5, 0)
    elseif TeleMode == "Tween" then
        local noclipConn
        noclipConn = game:GetService("RunService").Stepped:Connect(function()
            if lp.Character then
                for _, v in pairs(lp.Character:GetDescendants()) do
                    if v:IsA("BasePart") then v.CanCollide = false end
                end
            end
        end)
        
        local distance = (hrp.Position - targetCF.Position).Magnitude
        local speed = 150 
        local tweenInfo = TweenInfo.new(distance / speed, Enum.EasingStyle.Linear)
        local tween = game:GetService("TweenService"):Create(hrp, tweenInfo, {CFrame = targetCF + Vector3.new(0, 5, 0)})
        
        tween.Completed:Connect(function()
            if noclipConn then noclipConn:Disconnect() end
            for _, v in pairs(lp.Character:GetDescendants()) do
                if v:IsA("BasePart") then v.CanCollide = true end
            end
        end)
        tween:Play()
    end
end

local WebhookURL = ""
local lp = game.Players.LocalPlayer

local function SendWebhook(embedData)
    if WebhookURL == "" or not WebhookURL:find("discord.com/api/webhooks/") then return end
    local HttpService = game:GetService("HttpService")
    local request = syn and syn.request or http_request or request or HttpPost
    if request then
        request({
            Url = WebhookURL,
            Method = "POST",
            Headers = {["Content-Type"] = "application/json"},
            Body = HttpService:JSONEncode({["embeds"] = {embedData}})
        })
    end
end

local function GetStats()
    local beli = 0
    local dataFolder = lp:FindFirstChild("Data")
    if dataFolder and dataFolder:FindFirstChild("Belis") then
        beli = dataFolder.Belis.Value
    end
    local bounty = lp:FindFirstChild("leaderstats") and lp.leaderstats:FindFirstChild("Bounty") and lp.leaderstats.Bounty.Value or 0
    
    local items = {}
    for _, v in pairs(lp.Backpack:GetChildren()) do
        if v:IsA("Tool") then table.insert(items, "x1 " .. v.Name) end
    end
    return beli, bounty, #items > 0 and table.concat(items, "\n") or "Empty"
end

local function CreateEmbed(title, color, dropStatus, itemName, location)
    local beli, bounty, bp = GetStats()
    return {
        ["title"] = title,
        ["color"] = color,
        ["fields"] = {
            {["name"] = "💰 Belis:", ["value"] = "```\n" .. tostring(beli) .. "\n```", ["inline"] = true},
            {["name"] = "☠️ Bounty:", ["value"] = "```\n" .. tostring(bounty) .. "\n```", ["inline"] = true},
            {["name"] = "🎒 Backpack Hiện Tại:", ["value"] = "```\n" .. bp .. "\n```", ["inline"] = false},
            {["name"] = "📜 Info:", ["value"] = dropStatus .. "\n\nPlayer: **" .. lp.Name .. "**\nItem Vừa Rớt Tại " .. location .. ":\n```\n" .. itemName .. "\n```", ["inline"] = false},
            {["name"] = "⏰ Time:", ["value"] = os.date("%X %d/%m/%Y"), ["inline"] = false}
        },
        ["footer"] = {["text"] = "Shadow Hub • " .. os.date("%d/%m/%Y")}
    }
end

local isMonitoring = false
local function MonitorDrops()
    if isMonitoring then return end
    isMonitoring = true

    workspace.ChildAdded:Connect(function(child)
        task.wait(0.3)
        if child.Name:find("Fruit") or child.Name:find("Box") or child.Name:find("Book") then
            if child.Name ~= "HitBox" and child.Name ~= "Handle" and child.Name ~= "Part" then
                SendWebhook(CreateEmbed("Webhook Sea2 | Executer: Shadow Hub", 16711680, "BOSS DROP ITEM (MAP)", child.Name, "MAP"))
            end
        end
    end)

    lp.Backpack.ChildAdded:Connect(function(child)
        task.wait(0.3)
        if child:IsA("Tool") then
            if child.Name:find("Fruit") or child.Name:find("Box") or child.Name:find("Book") then
                SendWebhook(CreateEmbed("Webhook Sea2 | Executer: Shadow Hub", 65280, "ITEM RECEIVED IN BACKPACK", child.Name, "BACKPACK"))
            end
        end
    end)
end
